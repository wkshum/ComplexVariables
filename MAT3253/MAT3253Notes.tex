\section{Notation}


\noindent {\bf List of symbols and notation}

\begin{longtable}{ll}
 $i$ & Imaginary unit \\
 $\mathbb{N}$ &Set of integers larger than or equal to 1 \\
 $\mathbb{N}_{\geq 0}$& Set of integers larger than or equal to 0\\
 $\mathbb{Z}$ & Set of integers \\
 $\mathbb{Q}$ & Set of rational numbers \\
 $\mathbb{R}$ & Set of real numbers \\
 $\mathbb{C}$ & Set of complex numbers \\
 $\bar{\mathbb{C}}$ & Extended complex plane \\
 $\sup$, $\inf$ & Supremum and infinum \\
 $\limsup$ & Limit superior \\
 $\liminf$ & Limit inferior \\
 $\bar{z}$, $z^*$ & Conjugate of complex number $z$ \\
 $\real(z)$ & Real part of $z$ \\
 $\imag(z)$ & Imaginary part of $z$ \\
 $|z|$, $\abs(z)$ & The modulus/absolute value/magnitude of $z$ \\
 $\arg(z)$ & The argument/angle of $z$ \\
 $\Arg (z)$ & Principal argument of $z$ \\
 $[z_0,z_1,z_2,z_3]$ & Cross ratio \\
 $\infty$ &  Point at infinity \\
 $\exp(z)$ & Complex exponential function \\
 $\log(z)$ & Complex log function \\
 $\Log(z)$ & Principal branch of complex log function \\
 $f^{-1}(U)$ & Inverse image of a set $U$ through the function $f$\\
 $\overline{S}$ & Closure of a set $S$ in $\mathbb{C}$ \\
 $\overline{S}^X$ & Relative closure of $S$ in $X$ \\
 $\mathcal{T}_X$ & Relative topology of a set $X$ in $\mathbb{C}$ \\
 $\mathbf{1}_S$ & Indicator function of a set $S$ \\
 $D(z,r)$ & Open disk with radius $r$ centered at $z$ \\
 $\overline{D}(z,r)$ & Closed disk with radius $r$ centered at $z$ \\
 $(df)_\mathbf{p}$ & Differential of $f$ at a point $\mathbf{p}$\\
 $f(z) dz$ & Complex differential form \\
 $f'(z)$, $df/dz$ & Complex derivative of function $f$\\
 $f^{(n)}$ & the $n$-th order derivative of function $f(z)$ \\
 $\frac{\partial}{\partial x}$, $\frac{\partial }{\partial y}$ &  Partial differential operators with respect to $x$ and $y$\\
 $\frac{\partial}{\partial z}$, $\frac{\partial }{\partial \bar{z}}$ &  Partial differential operators with respect to $z$ and $\bar{z}$ \\
 $-C$ & Reverse of contour $C$ \\
 $C_1+C_2$ & Concatenation of contours $C_1$ and $C_2$ \\
 $\int_C f \, dz$ & Complex integral of $f$ along a curve $C$\\
 $\oint_C f \, dz$ & Complex integral of $f$ along a closed curve $C$\\ 
 $\Res(f;z_0)$ & Residue of function $f$ at the point $z_0$ \\
 $\mathcal{O}(D)$ & The ring of holomorphic functions defined on a domain $D$ \\
$\mathcal{M}(D)$ & The field of meromorphic functions defined on a domain $D$ \\
 $n(C;z_0)$ & Winding number of a closed curve $C$ relative to the point $z_0$ 
%$\wp(z;\Lambda)$ &  Weierstass P function
\end{longtable}
